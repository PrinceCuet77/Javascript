# Event Delegation

- Single event listener for multiple element &
- After adding a new element, that event listener is still working

```html
<!DOCTYPE html>
<html lang="en">
  <head>
    <meta charset="UTF-8" />
    <meta name="viewport" content="width=device-width, initial-scale=1.0" />
    <link rel="stylesheet" href="src/style.css" />
  </head>
  <body>
    <ul>
      <li>Apple</li>
      <li>Orange</li>
      <li>Kiwi</li>
    </ul>
    <br />
    <button>Add Pepe</button>

    <script src="src/script.js"></script>
  </body>
</html>
```

- JS file:

```js
const ul = document.querySelector('ul');
const btn = document.querySelector('button');

ul.addEventListener('click', function (event) {
  if (event.target.matches('li')) {
    const item = event.target;
    console.log('Selected', item.innerText);

    if (item.innerText === 'Apple') {
      item.style.color = 'red';
    } else if (item.innerText === 'Orange') {
      item.style.color = 'blue';
    } else if (item.innerText === 'Kiwi') {
      item.style.color = 'white';
    } else if (item.innerText === 'Pepe') {
      item.style.color = 'black';
    }
  }
});

btn.addEventListener('click', function () {
  const li = document.createElement('li');
  li.innerText = 'Pepe';
  ul.appendChild(li);
});
```
